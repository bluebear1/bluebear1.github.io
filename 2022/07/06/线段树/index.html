结构解析

线段树的目的是实现在logN时间内动态修改区间值和修改值,主要逻辑是把区间[1，N]划分成logN个层级，[2N,4N)个区间

  • 区间数据结构:每个区间这里由Node表示，不记录区间l,r，由[1,N] 二分获得，left,right分别代表左右二分子区间，
    val代表区间和，add代表子区间的每个树都需要+add，add大概是这个区间每个数都*了多少倍;
  • query:在[sl,sr]内查询 [l,r]范围内的区间和
  • update:在[sl,sr]内将[l,r]区间内每个元素+add
  • pushDown:需要切分父区间的时候（二分切分），动态创建子区间，并且根据区间长度和add设置好子区间val,add
  • pushUp:主要是用于更新的时候，[sl,sr]被切分成2N个子区间，每个子区间值计算完后，递归求和logN次

使用场景

区间修改求和，区间最大值，最小值，最大公因数等等

实战经验

Range模块

覆盖区间
Range模块是跟踪数字范围的模块。设计一个数据结构来跟踪表示为 半开区间 的范围并查询它们。
半开区间 [left, right) 表示所有 left <= x < right 的实数 x
实现 RangeModule 类:
RangeModule() 初始化数据结构的对象。
void addRange(int left, int right) 添加 半开区间 [left, right)，跟踪该区间中的每个实数。添加与当前跟踪的数字部分重叠的区间时，应当添加在区间 [left, right]中尚未跟踪的任何数字到该区间中。
boolean queryRange(int left, int right) 只有在当前正在跟踪区间 [left, right) 中的每一个实数时，才返回 true ，否则返回 false。
void removeRange(int left, int right) 停止跟踪 半开区间 [left, right) 中当前正在跟踪的每个实数

MyCalendar

区间累加
实现一个 MyCalendar 类来存放你的日程安排。如果要添加的日程安排不会造成 重复预订 ，则可以存储这个新的日程安排。
当两个日程安排有一些时间上的交叉时（例如两个日程安排都在同一时间内），就会产生 重复预订 。
日程可以用一对整数 start 和 end 表示，这里的时间是半开区间，即 [start, end), 实数 x 的范围为，  start <= x < end 。
实现 MyCalendar 类：
MyCalendar() 初始化日历对象。
boolean book(int start, int end) 如果可以将日程安排成功添加到日历中而不会导致重复预订，返回 true 。否则，返回 false 并且不要将该日程安排添加到日历中。

class MyCalendarThree {
class Node {
Node left, right;
int val, add;
}

    Node root;
    int N = (int) 1e9;

    public void pushUp(Node node) {
        node.val = Math.max(node.left.val, node.right.val);
    }

    public void pushDown(Node node) {
        if (node.left == null) node.left = new Node();
        if (node.right == null) node.right = new Node();
        node.left.val += node.add;
        node.right.val += node.add;
        node.left.add += node.add;
        node.right.add += node.add;
        node.add = 0;
    }

    public int query(Node node, int sl, int sr, int l, int r) {
        if (l <= sl && r >= sr) {
            return node.val;
        }
        int mid = sl + sr >> 1, ans = 0;
        pushDown(node);
        if (mid >= sl) ans = Math.max(ans, query(node.left, sl, mid, l, r));
        if (mid < sr) ans = Math.max(ans, query(node.right, mid + 1, sr, l, r));
        return ans;
    }

    public void update(Node node, int sl, int sr, int l, int r, int add) {
        if (l <= sl && r >= sr) {
            node.val += add;
            node.add += add;
            return;
        }
        int mid = sl + sr >> 1;
        pushDown(node);
        if (mid >= l) update(node.left, sl, mid, l, r, add);
        if (mid < r) update(node.right, mid + 1, sr, l, r, add);
        pushUp(node);
    }


    public MyCalendarThree() {
        root = new Node();
    }

    public int book(int start, int end) {
        update(root, 0, N, start, end - 1, 1);
        return root.val;
    }
}

class NumArray {
    class Node {
        Node left, right;
        int val, add;
    }

    Node root = new Node();
    int[] _nums;
    int n;

    public void pushUp(Node node) {
        node.val = node.left.val + node.right.val;
    }

    public void pushDown(Node node, int leftLen, int rightLen) {
        if (node.left == null) node.left = new Node();
        if (node.right == null) node.right = new Node();
        node.left.val += leftLen * node.add;
        node.right.val += rightLen * node.add;
        node.left.add = node.add;
        node.right.add = node.add;
        node.add = 0;
    }

    public void update(Node node, int sl, int sr, int l, int r, int add) {
        if (sl >= l && sr <= r) {
            node.val += (sr - sl + 1) * add;
            node.add += add;
            return;
        }
        int mid = sl + sr >> 1;
        pushDown(node, mid - sl + 1, sr - mid);
        if (l <= mid) update(node.left, sl, mid, l, r, add);
        if (r > mid) update(node.right, mid + 1, sr, l, r, add);
        pushUp(node);
    }

    public int query(Node node, int sl, int sr, int l, int r) {
        if (sl >= l && sr <= r) {
            return node.val;
        }
        int mid = sl + sr >> 1, ans = 0;
        pushDown(node, mid - sl + 1, sr - mid);
        if (l <= mid) ans += query(node.left, sl, mid, l, r);
        if (r > mid) ans += (node.right, mid + 1, sr, l, r);
        return ans;
    }

    public NumArray(int[] nums) {
        n = nums.length;
        _nums = nums;
    }

    public void update(int index, int val) {
        update(root, 0, n - 1, index, index, val - _nums[index]);
        _nums[index] = val;
    }

    public int sumRange(int left, int right) {
        return query(root,0,n-1,left,right);
    }
}

class Solution {
class Node {
Node left, right;
int val, add;
}

    Node root = new Node();
    int N = (int) 1e9;

    public void pushUp(Node node) {
        node.val = Math.max(node.left.val, node.right.val);
    }

    public void pushDown(Node node) {
        if (node.left == null) node.left = new Node();
        if (node.right == null) node.right = new Node();
        if(node.add!=0) {
            node.left.val = node.add;
            node.right.val = node.add;
            node.left.add = node.add;
            node.right.add = node.add;
            node.add = 0;
        }
    }

    public int query(Node node, int sl, int sr, int l, int r) {
        if (l <= sl && r >= sr) {
            return node.val;
        }
        int mid = sl + sr >> 1, ans = 0;
        pushDown(node);
        if (mid >= l) ans = Math.max(ans, query(node.left, sl, mid, l, r));
        if (mid < r) ans = Math.max(ans, query(node.right, mid + 1, sr, l, r));
        return ans;
    }

    public void update(Node node, int sl, int sr, int l, int r, int add) {
        if (sl >= l && sr <= r) {
            node.val = add;
            node.add = add;
            return;
        }
        int mid = sl + sr >> 1;
        pushDown(node);
        if (mid >= l) update(node.left, sl, mid, l, r, add);
        if (mid < r) update(node.right, mid + 1, sr, l, r, add);
        pushUp(node);
    }

    public List<Integer> fallingSquares(int[][] positions) {
        int n = positions.length;
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int h = query(root, 0, N, positions[i][0], positions[i][0] + positions[i][1] - 1);
            //覆盖写高度
            update(root, 0, N, positions[i][0], positions[i][0] + positions[i][1] - 1, h + positions[i][1]);
            ans.add(root.val);
        }
        return ans;
    }
}
